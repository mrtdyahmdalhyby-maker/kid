const noBtn = document.getElementById("no"); const messages = [ "صدك متحبني 😫", "راح اجي اخنكج 😂", "بطلت ماحبج 🗿" ]; let currentMessage = 0; function moveButton(){ document.querySelector("h2").innerText = messages[currentMessage]; currentMessage++; if(currentMessage >= messages.length){ currentMessage = 0; } const x = Math.random() * (window.innerWidth - 100); const y = Math.random() * (window.innerHeight - 60); noBtn.style.left = x + "px"; noBtn.style.top = y + "px"; } noBtn.addEventListener("mouseover", moveButton); noBtn.addEventListener("click", moveButton); document.getElementById("yes").onclick = () => { document.getElementById("result").innerHTML = "كيدي تحبك 😍❤️"; };
